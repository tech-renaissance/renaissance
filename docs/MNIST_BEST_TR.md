 * @version 1.0.0
 * @date 2026-05-30
 * @author Team Tech-Renaissance
 *

 * - 仅使用已验证可用的框架功能
 * - 在零框架改动前提下达到最高准确率
 *
 * 核心策略：
 * - 网络结构：1024→512→256（比AWY_FINAL_K更宽，补偿无Label Smoothing）
 * - 激活函数：ReLU（全票通过的最优选择）
 * - Bias：true（AWY_FINAL选择，补偿无BN）
 * - 优化器：AdamW with weight_decay=1e-4（中间值，平衡正则化）

#include "renaissance.h"
#include <chrono>
#include <iomanip>
#include <iostream>
#include <string>
#include <cstdlib>

using namespace tr;

enum class TestMode { CPU, GPU, AMP };

const char* mode_name(TestMode m) {
    switch (m) {
    case TestMode::CPU: return "CPU  [FP32]";
    case TestMode::GPU: return "GPU  [FP32]";
    case TestMode::AMP: return "AMP  [FP16]";
    default:            return "???";
    }
}

TestMode parse_cli(int argc, char** argv) {
    bool mode_set = false;
    TestMode mode = TestMode::GPU;

    for (int i = 1; i < argc; ++i) {
        std::string a = argv[i];
        if (a == "--cpu") {
            if (mode_set) { std::cerr << "Multiple mode flags.\n"; std::exit(1); }
            mode = TestMode::CPU;
            mode_set = true;
        } else if (a == "--gpu") {
            if (mode_set) { std::cerr << "Multiple mode flags.\n"; std::exit(1); }
            mode = TestMode::GPU;
            mode_set = true;
        } else if (a == "--amp") {
            if (mode_set) { std::cerr << "Multiple mode flags.\n"; std::exit(1); }
            mode = TestMode::AMP;
            mode_set = true;
        } else {
            std::cerr << "Usage: " << argv[0] << " --cpu | --gpu | --amp\n";
            std::exit(1);
        }
    }
    return mode;
}

int main(int argc, char** argv) {
    auto mode = parse_cli(argc, argv);

    switch (mode) {
    case TestMode::CPU:
        GLOBAL_SETTING.use_cpu();
        break;
    case TestMode::GPU:
        GLOBAL_SETTING.use_gpu("0").amp(false);
        break;
    case TestMode::AMP:
        GLOBAL_SETTING.use_gpu("0").amp(true);
        break;
    }

    GLOBAL_SETTING
        .manual_seed(42)           // 固定随机种子，确保可复现
        .local_batch_size(128)     // 标准batch size

        .val_resolution(28)

    PREPROCESSOR_SETTING
