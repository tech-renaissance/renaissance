# ============================================================================
# Top-Level Integration Tests (ResNet-50 Training)
# ============================================================================

message(STATUS "Configuring top-level integration tests")

# 设置C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 定义头文件路径
include_directories(
    ${CMAKE_SOURCE_DIR}/include
)

# 添加CUDA头文件路径（如果启用CUDA）
if(TR_USE_CUDA AND DEFINED TR_CUDA_PATH)
    include_directories(${TR_CUDA_PATH}/include)
endif()

# ============================================================================
# 创建演示可执行文件
# ============================================================================

# MLPerf Closed Division - ResNet-50
add_executable(resnet50_c resnet50_c.cpp)

# MLPerf Open Division - ResNet-50
add_executable(resnet50_o resnet50_o.cpp)

# 完全开放规则 - ResNet-50 (Fastest Mode)
add_executable(resnet50_f resnet50_f.cpp)

# MNIST 3-layer MLP (784-512-256-10)
add_executable(mnist_mlp_3 mnist_mlp_3.cpp)

# SimpleTask AXPY example: adaptive GPU configuration
add_executable(example_axpy example_axpy.cpp)

# SimpleTask AXPY example: forced CPU mode
add_executable(example_axpy_cpu example_axpy_cpu.cpp)

# SimpleTask dual-graph example: compute + transfer overlap
add_executable(example_dual_graph example_dual_graph.cpp)

# SimpleTask dual-graph example: compute + transfer on CPU
add_executable(example_dual_graph_cpu example_dual_graph_cpu.cpp)

# ============================================================================
# 设置输出目录
# ============================================================================

set_target_properties(resnet50_c PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(resnet50_o PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(resnet50_f PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(mnist_mlp_3 PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(example_axpy PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(example_axpy_cpu PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(example_dual_graph PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

set_target_properties(example_dual_graph_cpu PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin/tests/top
)

# ============================================================================
# Windows下为控制台程序设置子系统
# ============================================================================

if(WIN32)
    set_target_properties(resnet50_c PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(resnet50_o PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(resnet50_f PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(mnist_mlp_3 PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(example_axpy PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(example_axpy_cpu PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(example_dual_graph PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
    set_target_properties(example_dual_graph_cpu PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
endif()

# ============================================================================
# 编译选项
# ============================================================================

if(MSVC)
    # ResNet-50 Closed Division
    target_compile_options(resnet50_c PRIVATE /W4 /utf-8
        /wd4244  # Type conversion may lose data
        /wd4505  # Unreferenced function with internal linkage
        /wd4458  # Declaration hides class member
        /wd4127  # Conditional expression is constant
    )
    # ResNet-50 Open Division
    target_compile_options(resnet50_o PRIVATE /W4 /utf-8
        /wd4244  # Type conversion may lose data
        /wd4505  # Unreferenced function with internal linkage
        /wd4458  # Declaration hides class member
        /wd4127  # Conditional expression is constant
    )
    # ResNet-50 Fastest
    target_compile_options(resnet50_f PRIVATE /W4 /utf-8
        /wd4244  # Type conversion may lose data
        /wd4505  # Unreferenced function with internal linkage
        /wd4458  # Declaration hides class member
        /wd4127  # Conditional expression is constant
    )
    target_compile_options(mnist_mlp_3 PRIVATE /W4 /utf-8
        /wd4244  # Type conversion may lose data
        /wd4505  # Unreferenced function with internal linkage
        /wd4458  # Declaration hides class member
        /wd4127  # Conditional expression is constant
    )
else()
    target_compile_options(resnet50_c PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(resnet50_o PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(resnet50_f PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(mnist_mlp_3 PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(example_axpy PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(example_axpy_cpu PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(example_dual_graph PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
    target_compile_options(example_dual_graph_cpu PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-function)
endif()

# ============================================================================
# 设置日志级别（演示程序使用INFO级别，减少冗余输出）
# ============================================================================

target_compile_definitions(resnet50_c PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(resnet50_o PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(resnet50_f PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(mnist_mlp_3 PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(example_axpy PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(example_axpy_cpu PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(example_dual_graph PRIVATE TR_LOG_LEVEL=1)  # INFO级别
target_compile_definitions(example_dual_graph_cpu PRIVATE TR_LOG_LEVEL=1)  # INFO级别

# ============================================================================
# 链接renaissance统一库
# ============================================================================

target_link_libraries(resnet50_c PRIVATE renaissance)
target_link_libraries(resnet50_o PRIVATE renaissance)
target_link_libraries(resnet50_f PRIVATE renaissance)
target_link_libraries(mnist_mlp_3 PRIVATE renaissance)
target_link_libraries(example_axpy PRIVATE renaissance)
target_link_libraries(example_axpy_cpu PRIVATE renaissance)
target_link_libraries(example_dual_graph PRIVATE renaissance)
target_link_libraries(example_dual_graph_cpu PRIVATE renaissance)

# ============================================================================

# ============================================================================

setup_gpu_runtime_env(resnet50_c)
setup_gpu_runtime_env(resnet50_o)
setup_gpu_runtime_env(resnet50_f)
setup_gpu_runtime_env(mnist_mlp_3)
setup_gpu_runtime_env(example_axpy)
setup_gpu_runtime_env(example_axpy_cpu)
setup_gpu_runtime_env(example_dual_graph)
setup_gpu_runtime_env(example_dual_graph_cpu)
